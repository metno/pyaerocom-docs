for fil in files:
    mask = xr.open_dataset(fil)
    print(mask)
    bs_name = os.path.basename(fil)
    var = bs_name.split('.')
    print(var)
    _vars = mask.variables

    if 'source_code' in _vars:
        mask = mask.drop('source_code')

    if 'receptor_code' in _vars:
        print(mask.variables)
        mask = mask.drop('receptor_code')

    mask = mask.where(mask==1, int(0))
    #print(binary)
    enc_dict = {str(var[0]):{'zlib': 'True', 'compression_opts': 9}}
    #print(enc_dict)
    mask.to_netcdf(save_dir + bs_name, engine = 'h5netcdf', encoding = enc_dict)
